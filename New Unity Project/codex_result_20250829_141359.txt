    Python (OpenCV + skimage + scipy):

        * Install: pip install opencv-python-headless scikit-image scipy numpy

    Code:

        import cv2
        import numpy as np
        from skimage.morphology import skeletonize
        from scipy.interpolate import splprep, splev

        # load image (BGR)
        img = cv2.imread('maze.png')
        hsv = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)

        # green threshold - may need tuning
        lower = np.array([40, 100, 50])
        upper = np.array([90, 255, 255])
        mask = cv2.inRange(hsv, lower, upper)

        # clean mask
        kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5,5))
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel, iterations=2)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel, iterations=1)

        # skeletonize (skimage expects boolean image)
        bool_mask = mask > 0
        skel = skeletonize(bool_mask).astype(np.uint8) * 255

        # get skeleton points
        ys, xs = np.where(skel > 0)  # note: (row, col) => (y,x)
        pts = np.column_stack([xs, ys])

        # optionally order points along path (simple approach using
    distance-based sort from one end)
        # find endpoints (pixels with only one neighbor)
        from scipy.spatial import cKDTree
        tree = cKDTree(pts)
        # pick arbitrary start: left-most point
        start_idx = np.argmin(pts[:,0])
        # do greedy nearest-neighbor ordering
        order = [start_idx]
        used = set(order)
        while len(order) < len(pts):
            dists, idxs = tree.query(pts[order[-1]], k=10)  # look-ahead
            for i in idxs:
                if int(i) not in used:
                    order.append(int(i))
                    used.add(int(i))
                    break
            else:
                break
        ordered_pts = pts[order]

        # fit a spline through ordered_pts (downsample if large)
        sample = ordered_pts[::max(1, len(ordered_pts)//500)].T  # shape (2,N)
        tck, u = splprep(sample, s=5)  # s= smoothing param
        unew = np.linspace(0,1,200)
        out = splev(unew, tck)
        spline_pts = np.vstack(out).T  # (200,2)

        # draw spline back on image
        out_img = img.copy()
        for (x,y) in spline_pts.astype(int):
            cv2.circle(out_img, (x,y), 3, (0,0,255), -1)
        cv2.imwrite('maze_spline.png', out_img)

        # save waypoints
        np.savetxt('waypoints.csv', spline_pts, delimiter=',', header='x,y',
    comments='')
